328  inline void ComputeProportions() {
329  for (int y = 0; y < this->years; y++) {
330  for (int s = 0; s < this->seasons; s++) {
331 
332  variable total_sn;
333  variable total_sn_males;
334  variable total_sn_females;
335  variable& total_sn_b = this->survey_biomass_total[y * this->seasons + s];
336  variable& total_sn_b_males = this->survey_biomass_total_males[y * this->seasons + s];
337  variable& total_sn_b_females = this->survey_biomass_total_males[y * this->seasons + s];
338  total_sn_b = static_cast<REAL_T> (0.0);
339  variable total_n;
340  variable total_n_males;
341  variable total_n_females;
342  variable total_c;
343  variable total_c_males;
344  variable total_c_females;
345  variable& total_c_b = this->catch_biomass_total[y * this->seasons + s];
346  variable& total_c_b_males = this->catch_biomass_total[y * this->seasons + s];
347  variable& total_c_b_females = this->catch_biomass_total[y * this->seasons + s];
348  total_c_b = static_cast<REAL_T> (0.0);
349  size_t index = 0;
350  for (int a = 0; a <this->ages; a++) {
351  index = y * this->seasons * this->ages + (s) * this->ages + a;
352  total_sn += survey_numbers_at_age[index];
353  total_sn_males += survey_numbers_at_age_males[index];
354  total_sn_females += survey_numbers_at_age_females[index];
355 
356  total_sn_b += survey_biomass_at_age[index];
357  total_sn_b_males += survey_biomass_at_age_males[index];
358  total_sn_b_females += survey_biomass_at_age_females[index];
359 
360  total_n += numbers_at_age[index];
361  total_n_males += numbers_at_age_males[index];
362  total_n_females += numbers_at_age_females[index];
363 
364  total_c += catch_at_age[index];
365  total_c_males += catch_at_age_males[index];
366  total_c_females += catch_at_age_females[index];
367 
368  total_c_b += catch_biomass_at_age[index];
369  total_c_b_males += catch_biomass_at_age_males[index];
370  total_c_b_females += catch_biomass_at_age_females[index];
371  }
372 
373 
374  for (int a = 0; a <this->ages; a++) {
375  index = y * this->seasons * this->ages + (s) * this->ages + a;
376  survey_proportion_at_age[index] = survey_numbers_at_age[index] / total_sn;
377  survey_proportion_at_age_males[index] = survey_numbers_at_age_males[index] / total_sn_males;

383

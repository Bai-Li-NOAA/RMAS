78  void Prepare() {
79  if (estimating_recruitment_deviations) {
80  variable sum = static_cast<REAL_T> (0.0);
81  for (int i = 0; i < this->recruitment_deviations.size(); i++) {
82  sum += this->recruitment_deviations[i];
83  }
84 
85  for (int i = 0; i < this->recruitment_deviations.size(); i++) {
86  this->recruitment_deviations[i] -= sum / static_cast<REAL_T> (this->recruitment_deviations.size());
87  }

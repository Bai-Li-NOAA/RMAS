332  inline void ComputeProportions() {
333  for (int y = 0; y < this->years; y++) {
334  for (int s = 0; s < this->seasons; s++) {
335 
336  variable total_n;
337  variable total_n_males;
338  variable total_n_females;
339  variable total_c;
340  variable total_c_males;
341  variable total_c_females;
342  variable& total_c_b = this->catch_biomass_total[y * this->seasons + s];
343  variable& total_c_b_males = this->catch_biomass_total[y * this->seasons + s];
344  variable& total_c_b_females = this->catch_biomass_total[y * this->seasons + s];
345  total_c_b = static_cast<REAL_T> (0.0);
346  size_t index = 0;
347  for (int a = 0; a <this->ages; a++) {
348  index = y * this->seasons * this->ages + (s) * this->ages + a;
349 
350  total_n += numbers_at_age[index];
351  total_n_males += numbers_at_age_males[index];
352  total_n_females += numbers_at_age_females[index];
353 
354  total_c += catch_at_age[index];
355  total_c_males += catch_at_age_males[index];
356  total_c_females += catch_at_age_females[index];
357 
358  total_c_b += catch_biomass_at_age[index];
359  total_c_b_males += catch_biomass_at_age_males[index];
360  total_c_b_females += catch_biomass_at_age_females[index];
361  }
362 
363 
364  for (int a = 0; a <this->ages; a++) {
365  index = y * this->seasons * this->ages + (s) * this->ages + a;
366  proportion_at_age[index] = numbers_at_age[index] / total_n;
367  proportion_at_age_males[index] = numbers_at_age_males[index] / total_n_males;
368  proportion_at_age_females[index] = numbers_at_age_females[index] / total_n_females;
369 
370  catch_proportion_at_age[index] = catch_at_age[index] / total_c;
371  catch_proportion_at_age_males[index] = catch_at_age_males[index] / total_c_males;
372  catch_proportion_at_age_females[index] = catch_at_age_females[index] / total_c_females;
373 
374  catch_biomass_proportion_at_age[index] = catch_biomass_at_age[index] / total_c_b;
375  catch_biomass_proportion_at_age_males[index] = catch_biomass_at_age_males[index] / total_c_b_males;
376  catch_biomass_proportion_at_age_females[index] = catch_biomass_at_age_females[index] / total_c_b_females;
377 
378  }
379 
380  }
